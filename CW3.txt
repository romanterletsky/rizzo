# 2.  В інтервалі від 1 до 10 визначити числа
# •  парні, які діляться на 2,
# •  непарні, які діляться на 3,
# •  числа, які не діляться на 2 та 3.

#1
even = []
odd_three_division = []
not_three_not_two = []
for x in range(11):
    if x%2==0:
        even.append(x)
    elif x%2==1 and x%3==0:
        odd_three_division.append(x)
    elif x%2==1 and x%3==1:
        not_three_not_two.append(x)
print(even)
print(odd_three_division)
print(not_three_not_two)

#2
even = [x for x in range(11) if x%2==0]
odd_three_division = [x for x in range(11) if x%2==1 and x%3==0]
not_three_not_two = [x for x in range(11) if x%2==1 and x%3==1]
print (even)

print(not_three_not_two)

# 3.  Написати програму, яка обчислює факторіал числа, яке користувач вводить.(не використовувати рекурсивного виклику функції)

#1
x = int(input("Enter your num  want to get a factorial of, please: "))
fact = 1
for i in range(1, x+1):
    if x == 0:
        break
    fact*=i
print(fact)

#2
x = int(input("Enter a num you want to get a factorial of, please: "))
i = 1
fact = 1
while i <= x:
    if x == 0:
        break
    fact*=i
    i+=1
print(fact)

#                     11 is a prime number

#                     14 equals 2 * 7

       for i in range(2,x):
